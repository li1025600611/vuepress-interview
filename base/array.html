function unique(arr) {
  return Array.from(new Set(arr));
  // or return [...new Set(arr)]
}

// ES5
function unique2(arr) {
  for (let i = 0; i < arr.length; i++) {
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[j] === arr[i]) {
        arr.splice(j, 1);
        j--;
      }
    }
  }
  return arr;
}

console.log(unique(arr), unique2(arr));
